<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE log4j:configuration PUBLIC "-//APACHE//DTD LOG4J 1.2//EN" "log4j.dtd">
<!--
    https://github.com/spring-projects/spring-mvc-showcase/blob/master/src/main/resources/log4j.xml
-->
<log4j:configuration xmlns:log4j="http://jakarta.apache.org/log4j/">

    <!-- This appender will dump the logs on the tomcat console -->
    <appender name="console" class="org.apache.log4j.ConsoleAppender">
        <param name="Target" value="System.out"/>
        <layout class="org.apache.log4j.PatternLayout">
            <param name="ConversionPattern" value="%d{HH:mm:ss} [%t] %c{1} - %m%n"/>
        </layout>
    </appender>

    <appender name="appRollingFile" class="org.apache.log4j.RollingFileAppender">
        <param name="Threshold" value="debug"/>
        <param name="File" value="/opt/installs/tomcat/8.5.9/logs/csi/appLogs.log"/>
        <layout class="org.apache.log4j.PatternLayout">
            <param name="ConversionPattern" value="%d{HH:mm:ss} [%t] %c{1} - %m%n"/>
        </layout>
    </appender>

    <appender name="vendorLogsRollingFile" class="org.apache.log4j.DailyRollingFileAppender">
        <param name="Append" value="true"/>
        <param name="DatePattern" value="'.'yyyy-MM-dd"/>
        <param name="File" value="/opt/installs/tomcat/8.5.9/logs/csi/vendorLogs.log"/>
        <layout class="org.apache.log4j.PatternLayout">
            <param name="ConversionPattern" value="[%p] %d %c %M - %m%n"/>
        </layout>
    </appender>

    <logger name="org.springframework.core">
        <level value="debug"/>
    </logger>

    <logger name="org.springframework.beans">
        <level value="debug"/>
    </logger>

    <logger name="org.springframework.context">
        <level value="debug"/>
    </logger>

    <logger name="org.springframework.http">
        <level value="debug"/>
    </logger>

    <logger name="org.springframework.web">
        <level value="debug"/>
    </logger>

    <!--
        We define a appender here since we want it to go to a different appender that what is defined in the
        root logger
    -->
    <logger name="com.lingoace">
        <level value="debug"/>
        <appender-ref ref="appRollingFile"/>
    </logger>

    <!--
        All loggers inherit from root, so here, all debug or higher messages
        from all loggers will be printed to the vendorRollingFile appender.
    -->
    <root>
        <priority value="debug"/>
        <appender-ref ref="vendorLogsRollingFile"/>
    </root>

</log4j:configuration>
